
facil funcao titulo1 "Instalando o Apache"

#Instalar apache2 se nao existir
which apache2 || apt install -y apache2

#Permitir mudar o permalink
a2enmod rewrite

#Cache control
a2enmod expires
a2enmod headers

#Configurar firewall do apache
ufw allow in "Apache Full"
ufw allow 22/tcp

if [[ $(facil status ufw) == "active" ]] 
then
	ufw disable
	ufw --force enable
	echo Firewall ativado
else
	echo Firewall inativo
fi

#Verifica se est√° tudo certo com o apache
apache2ctl configtest

#Reiniciar o apache
service apache2 restart
